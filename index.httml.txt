<script>
function generarMatrices() {
  const filas = Number(document.getElementById("filas").value);
  const columnas = Number(document.getElementById("columnas").value);
  const cont = document.getElementById("contenedor");
  cont.innerHTML = "";

  cont.innerHTML += `<div class="matriz-title">Matriz A</div>`;
  cont.appendChild(crearTabla("A", filas, columnas));

  cont.innerHTML += `<div class="matriz-title">Matriz B</div>`;
  cont.appendChild(crearTabla("B", filas, columnas));
}

function crearTabla(nombre, f, c) {
  const div = document.createElement("div");
  for (let i = 0; i < f; i++) {
    for (let j = 0; j < c; j++) {
      div.innerHTML += `<input id="${nombre}_${i}_${j}" type="number" value="0">`;
    }
    div.innerHTML += "<br>";
  }
  return div;
}

function leerMatriz(nombre) {
  const filas = Number(document.getElementById("filas").value);
  const columnas = Number(document.getElementById("columnas").value);
  const M = [];
  for (let i = 0; i < filas; i++) {
    M[i] = [];
    for (let j = 0; j < columnas; j++) {
      M[i][j] = Number(document.getElementById(`${nombre}_${i}_${j}`).value);
    }
  }
  return M;
}

function sumar(A, B) {
  return A.map((fila, i) => fila.map((v, j) => v + B[i][j]));
}

function restar(A, B) {
  return A.map((fila, i) => fila.map((v, j) => v - B[i][j]));
}

function multiplicar(A, B) {
  const filasA = A.length, colsA = A[0].length;
  const filasB = B.length, colsB = B[0].length;
  if (colsA !== filasB) return "Dimensiones incompatibles";

  const R = Array.from({ length: filasA }, () => Array(colsB).fill(0));
  for (let i = 0; i < filasA; i++)
    for (let j = 0; j < colsB; j++)
      for (let k = 0; k < colsA; k++)
        R[i][j] += A[i][k] * B[k][j];
  return R;
}

function determinante(M) {
  if (M.length !== M[0].length) return "No es cuadrada";
  const n = M.length;

  if (n === 1) return M[0][0];
  if (n === 2) return M[0][0]*M[1][1] - M[0][1]*M[1][0];

  let det = 0;
  for (let j = 0; j < n; j++) {
    const sub = M.slice(1).map(fila => fila.filter((_, col) => col !== j));
    det += ((j % 2 === 0 ? 1 : -1) * M[0][j] * determinante(sub));
  }
  return det;
}

function mostrar(m) {
  if (typeof m === "number" || typeof m === "string") return m;
  return m.map(f => f.join("   ")).join("\n");
}

function operar(op) {
  const A = leerMatriz("A");
  const B = leerMatriz("B");
  let r;

  if (op === "sumar") r = sumar(A, B);
  if (op === "restar") r = restar(A, B);
  if (op === "multiplicar") r = multiplicar(A, B);
  if (op === "detA") r = determinante(A);
  if (op === "detB") r = determinante(B);

  document.getElementById("resultado").textContent = mostrar(r);
}
